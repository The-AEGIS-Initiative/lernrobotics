import React from 'react';

import MainPage from "./pages/MainPage";
import GetTest from "./pages/GetTest";
import PostTest from "./pages/PostTest";

import { Switch, Route } from 'react-router-dom';

import connectSocket from 'socket.io-client';
import { registerInitEvent } from './sockets/events.js';

/**
  * WebSocket connected to node.js server
  * https://socket.io/docs/emit-cheatsheet/
  * @memberof module:app
  * @example <caption>Example usage of socket</caption>
  * () => {
  * 	socket.emit("eventServerIsListeningTo")
  * }
  */
export const socket = connectSocket('http://localhost:8000');
registerInitEvent();

function App() {
  return (
	<Switch>
		<Route exact path="/" component={MainPage} />
		<Route exact path="/gettest" component={GetTest} />
		<Route exact path="/posttest" component={PostTest} />
	</Switch>
  );
}

export default App;
